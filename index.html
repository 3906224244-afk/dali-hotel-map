class BankAccount:
    def __init__(self, owner, balance=0):
        self.owner = owner
        self.balance = balance

    def deposit(self, amount):
        self.balance += amount
        return f"存入{amount}，余额{self.balance}"

    def withdraw(self, amount):
        if amount > self.balance:
            return f"余额不足！（当前余额{self.balance}）"
        self.balance -= amount
        return f"取出{amount}，余额{self.balance}"

    def transfer(self, amount, other_account):
        result = self.withdraw(amount)
        if "余额不足" not in result:
            other_account.deposit(amount)
            return f"转账{amount}给{other_account.owner}成功"
        return "转账失败"

    def __str__(self):
        return f"{self.owner}: ¥{self.balance}"

class Classroom:
    def __init__(self):
        self.students = []

    def add_student(self, student):
        self.students.append(student)

    def remove_student(self, sid):
        self.students = [s for s in self.students if s.sid != sid]

    def get_top_student(self):
        if not self.students:
            return None
        return max(self.students, key=lambda s: s.average())

    def class_average(self):
        if not self.students:
            return 0
        return sum(s.average() for s in self.students) / len(self.students)

import numpy as np

# 造简单二分类数据（只用iris前两类）
X = iris.data[:100, :2]  # (100, 2)
y = iris.target[:100]     # 0或1

# 初始化参数
w = np.zeros(2)
b = 0.0
lr = 0.1
epochs = 200

def sigmoid(z):
    return 1 / (1 + np.exp(-z))

losses = []
for epoch in range(epochs):
    # 前向传播
    z = X @ w + b
    y_pred = sigmoid(z)

    # 交叉熵损失
    eps = 1e-15  # 防止log(0)
    loss = -np.mean(y * np.log(y_pred + eps) +
                    (1 - y) * np.log(1 - y_pred + eps))
    losses.append(loss)

    # 梯度（逻辑回归的梯度推导结果）
    dw = (X.T @ (y_pred - y)) / len(y)
    db = np.mean(y_pred - y)

    # 更新参数
    w -= lr * dw
    b -= lr * db

    if epoch % 50 == 0:
        acc = np.mean((y_pred >= 0.5) == y)
        print(f"Epoch {epoch:3d}: loss={loss:.4f}, acc={acc:.3f}")

print(f"\n手写LR准确率: {np.mean((y_pred>=0.5)==y):.3f}")
print(f"sklearn LR准确率: {accuracy_score(y_test, y_pred):.3f}")

🎯 三个理解梯度下降的比喻

蒙眼下山：你站在山顶，蒙着眼，想到谷底。每走一步，你用脚探一探周围哪边最陡（梯度），朝最陡的方向走一步。重复，直到脚下是平的。

试水温：你想调到一个不烫不凉的水温。往热了扭一点，感受一下（前向传播）；太烫了，往反方向扭（负梯度）；每次扭的幅度越来越小（学习率衰减）。

射箭校准：射一箭（预测），看偏了多少（损失），调整瞄准（更新参数），再射一箭。重复到正中靶心。

import numpy as np
import matplotlib.pyplot as plt

# ====== Step 1: 造数据 ======
np.random.seed(42)
X = 2 * np.random.rand(100, 1)        # 100个点，范围0~2
true_w, true_b = 3.5, 1.2
y = true_w * X + true_b + np.random.randn(100, 1) * 0.3

plt.scatter(X, y, alpha=0.6)
plt.title("数据：y = 3.5x + 1.2 + 噪声")
plt.savefig("day03_data.png", dpi=150)

# ====== Step 2: 初始化参数 ======
w = np.random.randn()   # 随机猜一个值
b = np.random.randn()
lr = 0.01               # 学习率 —— 今天最重要的超参数
epochs = 500

m = len(X)  # 样本数
loss_history = []

# ====== Step 3: 训练循环 ======
for epoch in range(epochs):
    # 前向传播：用当前参数预测
    y_pred = w * X + b

    # 计算 MSE 损失
    loss = np.mean((y_pred - y) ** 2)
    loss_history.append(loss)

    # 计算梯度（MSE对w和b的偏导数）
    # 推导：∂loss/∂w = (2/m) * Σ((y_pred - y) * X)
    dw = (2 / m) * np.sum((y_pred - y) * X)   # w的梯度
    db = (2 / m) * np.sum(y_pred - y)          # b的梯度

    # 梯度下降：w ← w - lr * gradient
    w -= lr * dw
    b -= lr * db

    if epoch % 100 == 0:
        print(f"Epoch {epoch:3d} | Loss: {loss:.4f} | w: {w:.2f} | b: {b:.2f}")

print(f"\n✅ 训练完成！")
print(f"真实参数: w={true_w}, b={true_b}")
print(f"学到参数: w={w:.2f}, b={b:.2f}")
print(f"最终损失: {loss:.4f}")

# ====== Step 4: 三张图（损失曲线 + 拟合线 + 学习率对比）======
fig, axes = plt.subplots(1, 3, figsize=(15, 4))

# 左图：损失曲线（log尺度看得更清楚）
axes[0].plot(loss_history)
axes[0].set_xlabel("Epoch"); axes[0].set_ylabel("Loss")
axes[0].set_title("Loss 曲线 (会看到它一直在降)")
axes[0].set_yscale('log')

# 中图：拟合线 vs 数据

axes[1].plot(X, w * X + b, 'r-', linewidth=2,
             label=f"Fit: y={w:.2f}x+{b:.2f}")
axes[1].legend(); axes[1].set_title("拟合结果")

# 右图：学习率对比实验（等会补）

true_w = np.array([2.0, -1.5, 0.8])       # (3,)
y_multi = X_multi @ true_w + 0.5 + np.random.randn(100) * 0.5

# 初始化
w = np.random.randn(3)    # 现在是向量了！
b = np.random.randn()
lr = 0.01

for epoch in range(500):
    y_pred = X_multi @ w + b                # (100,)
    loss = np.mean((y_pred - y_multi) ** 2)

    # 梯度——矩阵形式（和单变量逻辑一样，只是维度不同）
    errors = y_pred - y_multi               # (100,)
    dw = (2 / m) * (X_multi.T @ errors)     # (3,) —— 这是关键！
    db = (2 / m) * np.sum(errors)           # 标量

    w -= lr * dw
    b -= lr * db

print(f"真实w: {true_w}")
print(f"学到w: {w}")
# w的三个维度和真实值应该非常接近

偏差-方差：理解过拟合的本质工具

x轴=模型复杂度，y轴=误差

偏差（Bias）：模型太简单，连训练数据都拟合不好。→ 欠拟合
方差（Variance）：模型太复杂，把训练数据的噪声都记住了。→ 过拟合
总误差 = 偏差² + 方差 + 不可约误差

好模型的复杂度 = 总误差最低的那个点。在纸上画出三条曲线（偏差/方差/总误差），标注"最佳复杂度"。

🧠 神经网络在做什么？

一个神经元 = 加权求和（Σwᵢxᵢ + b）→ 过激活函数 → 输出。一层神经元 = 多个神经元并行。多层堆叠 = 层层抽象（边缘→形状→物体）。

反向传播 = 链式法则的工程实现。从最后一层往前，逐层计算"每个参数对最终损失的贡献（梯度）"。Day 3你手算了dw和db——反向传播就是把这个过程自动化、系统化。

python 
import math

class Value:
    """计算图的基本单元——存数值 + 梯度 + 子节点"""
    def __init__(self, data, _children=(), _op='', label=''):
        self.data = data
        self.grad = 0.0
        self._backward = lambda: None
        self._prev = set(_children)
        self._op = _op
        self.label = label

    def __repr__(self):
        return f"Value(data={self.data:.4f}, grad={self.grad:.4f})"

    # 加法：梯度原样传递
    def __add__(self, other):
        other = other if isinstance(other, Value) else Value(other)
        out = Value(self.data + other.data, (self, other), '+')
        def _backward():
            self.grad += 1.0 * out.grad
            other.grad += 1.0 * out.grad
        out._backward = _backward
        return out

    # 乘法：d(a*b)/da = b, d(a*b)/db = a
    def __mul__(self, other):
        other = other if isinstance(other, Value) else Value(other)
        out = Value(self.data * other.data, (self, other), '*')
        def _backward():
            self.grad += other.data * out.grad
            other.grad += self.data * out.grad
        out._backward = _backward
        return out

    # tanh 激活函数
    def tanh(self):
        x = self.data
        t = (math.exp(2*x) - 1) / (math.exp(2*x) + 1)
        out = Value(t, (self,), 'tanh')
        def _backward():
            self.grad += (1 - t**2) * out.grad
        out._backward = _backward
        return out

    # 拓扑排序 + 反向传播
    def backward(self):
        topo = []
        visited = set()
        def build_topo(v):
            if v not in visited:
                visited.add(v)
                for child in v._prev:
                    build_topo(child)
                topo.append(v)
        build_topo(self)
        self.grad = 1.0
        for node in reversed(topo):
            node._backward()

import random

class Neuron:
    def __init__(self, nin):
        self.w = [Value(random.uniform(-1, 1)) for _ in range(nin)]
        self.b = Value(random.uniform(-1, 1))

    def __call__(self, x):
        act = sum((wi*xi for wi, xi in zip(self.w, x)), self.b)
        return act.tanh()

class Layer:
    def __init__(self, nin, nout):
        self.neurons = [Neuron(nin) for _ in range(nout)]

    def __call__(self, x):
        outs = [n(x) for n in self.neurons]
        return outs[0] if len(outs) == 1 else outs

class MLP:
    def __init__(self, nin, nouts):  # nouts = [4,4,1]
        sz = [nin] + nouts
        self.layers = [Layer(sz[i], sz[i+1]) for i in range(len(nouts))]

    def __call__(self, x):
        for layer in self.layers:
            x = layer(x)
        return x

from torch.utils.data import DataLoader
from torch.nn.utils.rnn import pad_sequence
from torchtext.datasets import IMDB
from torchtext.data.utils import get_tokenizer
from torchtext.vocab import build_vocab_from_iterator

tokenizer = get_tokenizer("basic_english")

def yield_tokens(data_iter):
    for label, text in data_iter:
        yield tokenizer(text)

vocab = build_vocab_from_iterator(yield_tokens(train_iter), specials=["",""])
vocab.set_default_index(vocab[""])

class LSTMClassifier(nn.Module):
    def __init__(self, vocab_size, embed_dim=128, hidden_dim=128, num_layers=2):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, embed_dim, padding_idx=vocab[""])
        self.lstm = nn.LSTM(embed_dim, hidden_dim, num_layers, batch_first=True, bidirectional=True)
        self.fc = nn.Linear(hidden_dim * 2, 2)  # *2 因为双向
        self.dropout = nn.Dropout(0.5)

    def forward(self, x):
        embedded = self.embedding(x)
        lstm_out, (h_n, c_n) = self.lstm(embedded)
        last = torch.cat((h_n[-2], h_n[-1]), dim=1)  # 双向的最后隐藏状态
        return self.fc(self.dropout(last))

Day 14 · Attention 机制 NumPy 实现

前置依赖：Day 13（Transformer理解）、Day -1（NumPy）

上午：纸面推导

在纸上推导 Scaled Dot-Product Attention，标注每一步的矩阵维度：

  1. Q = X @ W_Q → K = X @ W_K → V = X @ W_V
  2. scores = Q @ K^T → scaled = scores / √d_k
  3. weights = softmax(scaled) → output = weights @ V

为什么除以√d_k？防止点积过大导致softmax进入饱和区（梯度接近0）。

下午：NumPy从零实现

创建 day14_self_attention.py

python 
import numpy as np

seq_len, d_model = 4, 8
X = np.random.randn(seq_len, d_model)
W_Q = np.random.randn(d_model, d_model) * 0.1
W_K = np.random.randn(d_model, d_model) * 0.1
W_V = np.random.randn(d_model, d_model) * 0.1

Q, K, V = X @ W_Q, X @ W_K, X @ W_V

scores = (Q @ K.T) / np.sqrt(d_model)

def softmax(x):  # 减去max防溢出
    e = np.exp(x - np.max(x, axis=-1, keepdims=True))
    return e / np.sum(e, axis=-1, keepdims=True)

attn = softmax(scores)
output = attn @ V
print(f"Attention: {attn.shape} → Output: {output.shape}")

from langchain.document_loaders import TextLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.embeddings import HuggingFaceEmbeddings
from langchain.vectorstores import Chroma
from langchain.chains import RetrievalQA
from langchain.llms import Ollama  # 本地免费模型

# 1. 加载你的Obsidian笔记
loader = TextLoader("../notes/day01_review.md")
docs = loader.load()

# 2. 切分（chunk_size是核心超参数）
splitter = RecursiveCharacterTextSplitter(chunk_size=500, chunk_overlap=50)
chunks = splitter.split_documents(docs)

# 3. 向量化 + 存入ChromaDB
embeddings = HuggingFaceEmbeddings(model_name="all-MiniLM-L6-v2")
vectorstore = Chroma.from_documents(chunks, embeddings)

# 4. 检索 + 问答
llm = Ollama(model="qwen2.5:7b")  # 或 llama3.1
qa = RetrievalQA.from_chain_type(llm=llm, retriever=vectorstore.as_retriever())

print(qa.run("什么是梯度下降？"))
